def to_us

	accents = { ['à','â','ä'] => 'a',['Ä','Â','À'] => 'A',
	['é','è','ê','ë'] => 'e', ['Ë','É','È','Ê'] => 'E',
	['î','ï'] => 'i', ['Ï','Î'] => 'I',
	['ô','ö'] => 'o', ['Ö','Ô'] => 'O',
	['ù','û','ü'] => 'u', ['Ü','Û','Ù'] => 'U',
	['ç'] => 'c', ['Ç'] => 'C'}
	str = self.to_s

	accents.each do |accent, echange|
		accent.each do |c|
			str.gsub!(c,echange)
		end
	
	end
	return str
end
