def insert(elemN:)
	if(elemN < elem)
		if(elemG.nil?)
			@elemG = Arbre.new(elem: elemN)
		else
			@elemG.insert(elemN: elemN)
		end
	end
	
	if(elemN > elem)
		if(elemD.nil?)
			@elemD = Arbre.new(elem: elemN)
		else
			@elemD.insert(elemN: elemN)
		end
	end
	
	if(elemN == elem)
		raise "#{elem} est déjà présent dans l'arbre !"
	end
end
